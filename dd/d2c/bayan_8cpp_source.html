122 dirs.begin(),
123 dirs.end(),
124 std::inserter(res_dirs, res_dirs.end()),

137 if (scanned_paths.count(path) > 0) {
138 return;
139 }
140
141 scanned_paths.insert(path);
142
143 if (!std::filesystem::exists(path)) {
144 return;
145 }
146
147 for (const auto& ignore_dir : ignore_dirs) {
148 auto relative_ignore = std::filesystem::relative(ignore_dir, path);
149 if (relative_ignore.empty() || relative_ignore.native()[0] == '.') {
150 return;

152 }
153

155 auto relative = std::filesystem::relative(path, current_path);
156
157 std::size_t d = 0;
158 for (const auto& _ : relative) {
159 if (d++ >= scan_depth_.value()) {
160 return;
161 }
162 }
163 }
164
165 std::smatch base_match;
166
167 for (const auto& dir_entry : std::filesystem::directory_iterator(path)) {

169 continue;
170 }
171
172 const auto& entry_path = dir_entry.path();
173 if (scanned_paths.count(entry_path) > 0) {
174 continue;
175 }
176
177 scanned_paths.insert(entry_path);
178
179 if (dir_entry.is_regular_file()) {
180 if (std::none_of(masks.begin(), masks.end(), [&](const std::regex& r) {
181 return std::regex_search(entry_path.string(), r);
182 })) {
183 continue;
184 }
185
186 auto file_size = static_cast<std::size_t>(dir_entry.file_size());
187 if (file_size >= min_file_size_) {
188 file_infos[file_size].insert(entry_path);
189 }
190 } else {
191 scan_dirs(entry_path);
192 }
193 }
194 };
195
196 for (const auto& dir_to_scan : dirs_to_scan) {
197 scan_dirs(dir_to_scan);
198 }
199
200 for (auto it = file_infos.begin(); it != file_infos.end();) {
201 if (it->second.size() < 2) {
202 it = file_infos.erase(it);
203 } else {
204 ++it;
205 }
206 }

213std::optional<HashValue> ScanChunk(
214 const std::filesystem::path& path,
215 std::size_t block_size,
216 std::size_t block_index,
217 const HashFunction& hasher) {
218 if (block_size == 0) {
219 return std::nullopt;
220 }
221
222 std::ifstream is{path, std::ios::in | std::ios::binary | std::ios::ate};
223 std::size_t file_size = static_cast<std::size_t>(is.tellg());
224 std::size_t cur_pos = block_size * block_index;
225 if (cur_pos >= file_size) {
226 return std::nullopt;
227 }
228
229 std::vector<char> chunk(block_size);
230 is.seekg(cur_pos, is.beg);
231 std::size_t chunk_end = block_size * (block_index + 1);
232 std::size_t read_len = cur_pos + block_size > file_size ? file_size - cur_pos : block_size;
233 is.read(chunk.data(), read_len);
234
235 return hasher(chunk);
236}

238void ScanChunkInfo(
239 FileChunkInfo& chunk_info,
240 std::size_t block_size,
241 std::size_t block_index,
242 const HashFunction& hasher) {
243 auto& child_chunks = chunk_info.child_chunks;
244
245 for (const auto& path : chunk_info.paths) {
246 auto hash_info = ScanChunk(path, block_size, block_index, hasher);
247 if (!hash_info) {

251 auto& child_chunk = child_chunks[hash_info.value()];
252 child_chunk.paths.insert(path);
